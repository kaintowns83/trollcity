import { createClientFromRequest } from 'npm:@base44/sdk@0.7.1';

Deno.serve(async (req) => {
  try {
    const base44 = createClientFromRequest(req);
    
    const user = await base44.auth.me();
    if (!user) {
      return Response.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const { streamId, answer } = await req.json();

    if (!streamId || !answer) {
      return Response.json({ error: 'Stream ID and answer required' }, { status: 400 });
    }

    console.log('ğŸ“¡ Storing WebRTC answer for stream:', streamId, 'from user:', user.id);

    // Get existing participant
    const participants = await base44.entities.StreamParticipant.filter({
      stream_id: streamId,
      user_id: user.id
    });

    if (participants[0]) {
      // Store answer in participant record
      await base44.entities.StreamParticipant.update(participants[0].id, {
        webrtc_answer: answer
      });
    }

    return Response.json({
      success: true,
      message: 'Answer stored successfully'
    });

  } catch (error) {
    console.error('âŒ Error storing WebRTC answer:', error);
    return Response.json({
      error: error.message
    }, { status: 500 });
  }
});
