import { createClientFromRequest } from 'npm:@base44/sdk@0.7.1';

Deno.serve(async (req) => {
  try {
    const base44 = createClientFromRequest(req);
    
    const user = await base44.auth.me();
    if (!user) {
      return Response.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const { streamId } = await req.json();

    if (!streamId) {
      return Response.json({ error: 'Stream ID required' }, { status: 400 });
    }

    console.log('ğŸ“¡ Getting WebRTC offer for stream:', streamId);

    const streams = await base44.entities.Stream.filter({ id: streamId });
    const stream = streams[0];

    if (!stream) {
      return Response.json({ error: 'Stream not found' }, { status: 404 });
    }

    return Response.json({
      success: true,
      offer: stream.webrtc_offer || null
    });

  } catch (error) {
    console.error('âŒ Error getting WebRTC offer:', error);
    return Response.json({
      error: error.message
    }, { status: 500 });
  }
});
